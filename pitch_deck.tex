% SECTION 7: KEY USER JOURNEYS
\section{User Journeys}

\begin{frame}{Journey 1: New Member Registration}
\begin{enumerate}
\item Visit \texttt{karumaincu.org/register}
\item Fill personal details (name, email, phone)
\item Enter student info (reg number, course, year)
\item Choose member type (student/associate)
\item Agree to doctrinal statement
\item Submit registration
\item Account created (status: pending)
\item Admin reviews and approves
\item Member receives access
\item Login and explore member portal
\end{enumerate}

\vspace{0.5em}
\centering
\textbf{Time: 2-3 minutes | Approval: 24 hours}
\end{frame}
